    function openDB() {
      return new Promise((resolve, reject) => {
        const request = indexedDB.open(DB_NAME, 1);

        request.onupgradeneeded = function(event) {
          const database = event.target.result;
          if (!database.objectStoreNames.contains(RECORD_STORE)) {
            database.createObjectStore(RECORD_STORE, { keyPath: "id" });
          }
          if (!database.objectStoreNames.contains(NAME_STORE)) {
            database.createObjectStore(NAME_STORE, { keyPath: "id" });
          }
        };

        request.onsuccess = function(event) {
          resolve(event.target.result);
        };

        request.onerror = function(event) {
          reject(event.target.error);
        };
      });
    }

    function putItem(storeName, item) {
      return new Promise((resolve, reject) => {
        const tx = db.transaction(storeName, "readwrite");
        const store = tx.objectStore(storeName);
        store.put(item);
        tx.oncomplete = () => resolve();
        tx.onerror = () => reject(tx.error);
      });
    }

    function getItem(storeName, id) {
      return new Promise((resolve, reject) => {
        const tx = db.transaction(storeName, "readonly");
        const store = tx.objectStore(storeName);
        const request = store.get(id);
        request.onsuccess = () => resolve(request.result || null);
        request.onerror = () => reject(request.error);
      });
    }

    function getAllItems(storeName) {
      return new Promise((resolve, reject) => {
        const tx = db.transaction(storeName, "readonly");
        const store = tx.objectStore(storeName);
        const request = store.getAll();
        request.onsuccess = () => resolve(request.result || []);
        request.onerror = () => reject(request.error);
      });
    }

    function deleteItem(storeName, id) {
      return new Promise((resolve, reject) => {
        const tx = db.transaction(storeName, "readwrite");
        const store = tx.objectStore(storeName);
        store.delete(id);
        tx.oncomplete = () => resolve();
        tx.onerror = () => reject(tx.error);
      });
    }

    function clearStore(storeName) {

    function roundedRect(ctx, x, y, w, h, r, fillStyle, strokeStyle, lineWidth = 0) {
      ctx.beginPath();
      ctx.moveTo(x + r, y);
      ctx.arcTo(x + w, y, x + w, y + h, r);
      ctx.arcTo(x + w, y + h, x, y + h, r);
      ctx.arcTo(x, y + h, x, y, r);
      ctx.arcTo(x, y, x + w, y, r);
      ctx.closePath();
      if (fillStyle) {
        ctx.fillStyle = fillStyle;
        ctx.fill();
      }
      if (strokeStyle && lineWidth > 0) {
        ctx.lineWidth = lineWidth;
        ctx.strokeStyle = strokeStyle;
        ctx.stroke();
      }
    }

    function drawFlowerCanvas(ctx, centerX, topY, name) {
      const petalColor = "#ffd3df";
      const petalColor2 = "#ffdce6";
      const centerColor = "#ffd54f";
      const stemColor = "#3c8a4a";
      const leafColor = "#7cc66a";
      const soilColor = "#ad7a45";

      // petals
      const petals = [
        [0, -30, petalColor],
        [30, 0, petalColor],
        [0, 30, petalColor],
        [-30, 0, petalColor],
        [22, -22, petalColor2],
        [-22, 22, petalColor2],
      ];
      for (const [dx, dy, color] of petals) {
        ctx.beginPath();
        ctx.fillStyle = color;
        ctx.arc(centerX + dx, topY + 44 + dy, 24, 0, Math.PI * 2);
        ctx.fill();
      }

      // center
      ctx.beginPath();
      ctx.fillStyle = centerColor;
      ctx.arc(centerX, topY + 44, 20, 0, Math.PI * 2);
      ctx.fill();
      ctx.fillStyle = "#7a4a00";
      ctx.font = "bold 18px Pretendard, Noto Sans KR, sans-serif";
      ctx.textAlign = "center";
      ctx.textBaseline = "middle";
      ctx.fillText("♥", centerX, topY + 45);

      // stem
      ctx.fillStyle = stemColor;
      roundedRect(ctx, centerX - 4, topY + 76, 8, 54, 8, stemColor, null, 0);

      // leaves
      ctx.save();
      ctx.translate(centerX - 28, topY + 94);
      ctx.rotate(-0.45);
      ctx.fillStyle = leafColor;
      ctx.beginPath();
      ctx.ellipse(0, 0, 18, 10, 0, 0, Math.PI * 2);
      ctx.fill();
      ctx.restore();

      ctx.save();
      ctx.translate(centerX + 28, topY + 94);
      ctx.rotate(0.45);
      ctx.fillStyle = leafColor;
      ctx.beginPath();
      ctx.ellipse(0, 0, 18, 10, 0, 0, Math.PI * 2);
      ctx.fill();
      ctx.restore();

      // soil
      ctx.beginPath();
      ctx.fillStyle = soilColor;
      ctx.ellipse(centerX, topY + 132, 42, 12, 0, 0, Math.PI * 2);
      ctx.fill();

      // label
      roundedRect(ctx, centerX - 45, topY + 138, 90, 28, 14, "#ffffff", "#d7c6a5", 2);
      ctx.fillStyle = "#3b3b3b";
      ctx.font = "bold 13px Pretendard, Noto Sans KR, sans-serif";
      ctx.fillText(name || "꽃", centerX, topY + 152);
    }

    function audioBufferToWavBlob(buffer) {
      const numChannels = buffer.numberOfChannels;
      const sampleRate = buffer.sampleRate;
      const format = 1;
      const bitDepth = 16;

      let result;
      if (numChannels === 1) {
        result = buffer.getChannelData(0);
      } else {
        const length = buffer.length * numChannels;
        result = new Float32Array(length);
        const channels = [];
        for (let i = 0; i < numChannels; i++) {
          channels.push(buffer.getChannelData(i));
        }
        let offset = 0;
        for (let i = 0; i < buffer.length; i++) {
          for (let ch = 0; ch < numChannels; ch++) {
            result[offset++] = channels[ch][i];
          }
        }
      }

      const bytesPerSample = bitDepth / 8;
      const blockAlign = numChannels * bytesPerSample;
      const bufferLength = 44 + result.length * bytesPerSample;
      const arrayBuffer = new ArrayBuffer(bufferLength);
      const view = new DataView(arrayBuffer);

      function writeString(view, offset, string) {
        for (let i = 0; i < string.length; i++) {
          view.setUint8(offset + i, string.charCodeAt(i));
        }
      }

      let offset = 0;
      writeString(view, offset, "RIFF"); offset += 4;
      view.setUint32(offset, 36 + result.length * bytesPerSample, true); offset += 4;
      writeString(view, offset, "WAVE"); offset += 4;
      writeString(view, offset, "fmt "); offset += 4;
      view.setUint32(offset, 16, true); offset += 4;
      view.setUint16(offset, format, true); offset += 2;
      view.setUint16(offset, numChannels, true); offset += 2;
      view.setUint32(offset, sampleRate, true); offset += 4;
      view.setUint32(offset, sampleRate * blockAlign, true); offset += 4;
      view.setUint16(offset, blockAlign, true); offset += 2;
      view.setUint16(offset, bitDepth, true); offset += 2;
      writeString(view, offset, "data"); offset += 4;
      view.setUint32(offset, result.length * bytesPerSample, true); offset += 4;

      let index = 44;
      for (let i = 0; i < result.length; i++, index += 2) {
        let s = Math.max(-1, Math.min(1, result[i]));
        view.setInt16(index, s < 0 ? s * 0x8000 : s * 0x7FFF, true);
      }

      return new Blob([view], { type: "audio/wav" });
    }

    async function saveAllAudio() {
      const recordings = await getAllRecordings();
      if (recordings.length !== TOTAL_FLOWERS) {
        alert("16개 꽃이 모두 피어난 뒤에 전체 음성을 저장할 수 있어요.");
        return;
      }

      const sorted = recordings.sort((a, b) => getFlowerNumber(a.id) - getFlowerNumber(b.id));
      const AudioContextClass = window.AudioContext || window.webkitAudioContext;
      if (!AudioContextClass) {
        alert("이 브라우저에서는 전체 음성 저장을 지원하지 않아요.");
        return;
      }

      const audioContext = new AudioContextClass();
      const decodedBuffers = [];

      for (const item of sorted) {
        const arrayBuffer = await item.blob.arrayBuffer();
        const decoded = await audioContext.decodeAudioData(arrayBuffer.slice(0));
        decodedBuffers.push(decoded);
      }

      if (decodedBuffers.length === 0) {
        alert("저장할 음성이 없어요.");
        return;
      }

      const sampleRate = decodedBuffers[0].sampleRate;
      const channels = Math.max(...decodedBuffers.map(buf => buf.numberOfChannels));
      const totalLength = decodedBuffers.reduce((sum, buf) => sum + buf.length, 0);

      const offline = new OfflineAudioContext(channels, totalLength, sampleRate);
      let offset = 0;

      for (const buf of decodedBuffers) {
        const source = offline.createBufferSource();
        source.buffer = buf;
        source.connect(offline.destination);
        source.start(offset / sampleRate);
        offset += buf.length;
      }

      const rendered = await offline.startRendering();
      const wavBlob = audioBufferToWavBlob(rendered);
      downloadBlob(wavBlob, getClassFileName("나눔_꽃밭_전체_음성.wav"));

      if (audioContext.close) {
        await audioContext.close();
      }
    }

    function startTimer() {
      seconds = 0;
      timerEl.textContent = "00:00";

        timerEl.textContent = `${min}:${sec}`;
      }, 1000);

    async function startRecording() {
      if (!selectedFlower) {
        statusEl.textContent = "먼저 꽃을 골라주세요.";
        return;
      }

      if (nameInput.value.trim()) {
        await saveName(selectedFlower, nameInput.value.trim());
      }

      try {
        stream = await navigator.mediaDevices.getUserMedia({ audio: true });
        audioChunks = [];
        mediaRecorder = new MediaRecorder(stream);

        mediaRecorder.addEventListener("dataavailable", event => {
          if (event.data.size > 0) {
            audioChunks.push(event.data);
          }
        });

        mediaRecorder.addEventListener("stop", async () => {
          const mimeType = mediaRecorder.mimeType || "audio/webm";
          const audioBlob = new Blob(audioChunks, { type: mimeType });

          await saveRecording(selectedFlower, audioBlob);

          if (stream) {
            stream.getTracks().forEach(track => track.stop());
            stream = null;
          }

          const number = getFlowerNumber(selectedFlower);
          const name = await getName(selectedFlower);

          mediaRecorder = null;
          stopTimer();

          statusEl.textContent = `${name || number + "번 꽃"}에 나눔 말이 저장되었어요. 꽃이 활짝 피어났어요! 이제 꽃을 누르면 바로 목소리가 들려요.`;
          await renderGarden();
          await selectFlower(number);

          const savedRecordings = await getAllRecordings();
          if (savedRecordings.length === TOTAL_FLOWERS) {
            showGallery();
          }
        });

        mediaRecorder.start();
        startTimer();
        statusEl.textContent = "녹음 중이에요. 아이의 나눔 말을 말한 뒤 녹음 끝을 눌러주세요.";
        await updateButtons();

      } catch (error) {
        console.error(error);
        statusEl.textContent = "마이크 권한이 필요해요. 브라우저에서 마이크 사용을 허용해주세요.";
      }
    }

    function stopRecording() {

      }
    }

    async function playFlower(id) {
      const saved = await getRecording(id);
      if (!saved) {
        statusEl.textContent = "아직 이 꽃에는 녹음된 나눔 말이 없어요.";
        return;
      }

      if (currentAudio) {
        currentAudio.pause();
        currentAudio = null;
      }

      document.querySelectorAll(".flower.playing").forEach(el => el.classList.remove("playing"));
      document.querySelectorAll(`[data-id="${id}"]`).forEach(el => el.classList.add("playing"));

      const name = await getName(id);
      const number = getFlowerNumber(id);
      const audioUrl = URL.createObjectURL(saved.blob);
      const audio = new Audio(audioUrl);
      currentAudio = audio;

      const message = `${name || number + "번 꽃"}의 나눔 말을 듣고 있어요.`;
      statusEl.textContent = message;
      galleryMessage.textContent = message;

      audio.play();

      audio.onended = () => {
        URL.revokeObjectURL(audioUrl);
        document.querySelectorAll(`[data-id="${id}"]`).forEach(el => el.classList.remove("playing"));
        currentAudio = null;
      };

      audio.onerror = () => {
        URL.revokeObjectURL(audioUrl);
        document.querySelectorAll(`[data-id="${id}"]`).forEach(el => el.classList.remove("playing"));
        currentAudio = null;
      };
    }

        statusEl.textContent = "아직 피어난 꽃이 없어요.";
        galleryMessage.textContent = "아직 피어난 꽃이 없어요.";
        return;
      }

      statusEl.textContent = `피어난 꽃 ${sorted.length}송이의 나눔 말을 차례대로 들려줄게요.`;
      galleryMessage.textContent = statusEl.textContent;

      for (const item of sorted) {
        await new Promise(async resolve => {
          const name = await getName(item.id);
          document.querySelectorAll(".flower.playing").forEach(el => el.classList.remove("playing"));
          document.querySelectorAll(`[data-id="${item.id}"]`).forEach(el => el.classList.add("playing"));

          const audioUrl = URL.createObjectURL(item.blob);
          const audio = new Audio(audioUrl);

          const msg = `${name || getFlowerNumber(item.id) + "번 꽃"}의 나눔 말을 듣고 있어요.`;
          statusEl.textContent = msg;
          galleryMessage.textContent = msg;

          audio.onended = () => {
            URL.revokeObjectURL(audioUrl);
            document.querySelectorAll(`[data-id="${item.id}"]`).forEach(el => el.classList.remove("playing"));
            setTimeout(resolve, 500);
          };

          audio.onerror = () => {
            URL.revokeObjectURL(audioUrl);
            document.querySelectorAll(`[data-id="${item.id}"]`).forEach(el => el.classList.remove("playing"));
            resolve();
          };

          audio.play();
        });
      }

      statusEl.textContent = "피어난 꽃들의 나눔 말을 모두 들었어요.";
      galleryMessage.textContent = "피어난 꽃들의 나눔 말을 모두 들었어요.";
    }
